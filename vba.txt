Sub AtualizarInadimplencia()
    'carregar os dados na memória do Excel
    Dim wsOrigem As Worksheet, wsInad As Worksheet
    Dim ultLinhaOrigem As Long, ultLinhaMain As Long
    Dim qtdOrigem As Long, qtdInad As Long
    Dim diff As Long
    
    'apelidos das abas
    Set wsOrigem = ThisWorkbook.Sheets("Origem dados")
    Set wsInad = ThisWorkbook.Sheets("Inadimplencia")
    
    
    'pausa a atualização de tela do excel temporariamente
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    
   'Contagem de linhas da Origem
    ultLinhaOrigem = wsOrigem.Cells(wsOrigem.Rows.Count, "A").End(xlUp).Row
    qtdOrigem = ultLinhaOrigem - 1
    
    'se a quantiadade de linhas da origem for menor que 1
    If qtdOrigem < 1 Then
        Application.Calculation = xlCalculationAutomatic
        Application.ScreenUpdating = True
        MsgBox "A Origem de dados está vazia! Nenhuma atualização foi feita.", vbExclamation, "Aviso"
        Exit Sub
    End If
    
    'Contagem de linhas da Inadimplencia
    ultLinhaMain = wsInad.Cells(wsInad.Rows.Count, "A").End(xlUp).Row
    If ultLinhaMain < 2 Then ultLinhaMain = 2
    qtdInad = ultLinhaMain - 1
    
    
'''''''
    If qtdOrigem > qtdInad Then
        diff = qtdOrigem - qtdInad
        wsInad.Rows((ultLinhaMain + 1) & ":" & (ultLinhaMain + diff)).Insert Shift:=xlDown
        wsInad.Range("A" & ultLinhaMain & ":M" & ultLinhaMain).Copy
        wsInad.Range("A" & (ultLinhaMain + 1) & ":M" & (ultLinhaMain + diff)).PasteSpecial Paste:=xlPasteFormulas
    ElseIf qtdOrigem < qtdInad And qtdOrigem > 0 Then
        wsInad.Rows((ultLinhaOrigem + 1) & ":" & ultLinhaMain).Delete Shift:=xlUp
    End If
'''''''
    ultLinhaMain = qtdOrigem + 1

    'limpeza de cache
    wsInad.Range("A2:A" & ultLinhaMain).ClearContents
    wsInad.Range("D2:D" & ultLinhaMain).ClearContents
    wsInad.Range("G2:G" & ultLinhaMain).ClearContents
    
    'Cola os dados da Origem como Valores para não levar a formatação da base de Origem junto
    wsOrigem.Range("A2:A" & ultLinhaOrigem).Copy
    wsInad.Range("A2").PasteSpecial Paste:=xlPasteValues
    wsOrigem.Range("B2:B" & ultLinhaOrigem).Copy
    wsInad.Range("D2").PasteSpecial Paste:=xlPasteValues
    wsOrigem.Range("C2:C" & ultLinhaOrigem).Copy
    wsInad.Range("G2").PasteSpecial Paste:=xlPasteValues
    
    'ligando a atualização automátia para os PROCVs atualizarem
    Application.Calculation = xlCalculationAutomatic
    
    
    wsInad.Sort.SortFields.Clear 'limpando cache de memórias anteriores
    
    ' Ordenar por Vendedor e depois por Cidade
    wsInad.Range("A1:M" & ultLinhaMain).Sort _
        Key1:=wsInad.Range("K1"), Order1:=xlAscending, _
        Key2:=wsInad.Range("C1"), Order2:=xlAscending, _
        Header:=xlYes, _
        Orientation:=xlTopToBottom
        
    'A tela volta ao normal e limpa a área de transferência
    Application.CutCopyMode = False
    Application.ScreenUpdating = True
    
    ' Mensagem final
    MsgBox "Atualizado e ordenado por VENDEDOR e CIDADE!", vbInformation, "Concluído"
End Sub
